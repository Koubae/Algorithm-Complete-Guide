func inorderTraversal(root *Node) []int {
	var algo func(node *Node)

	result := make([]int, 0)
	algo = func(node *Node) {
		if node == nil {
			return
		}

		algo(node.Left)
		result = append(result, node.Val)
		algo(node.Right)

	}
	algo(root)
	return result
}
